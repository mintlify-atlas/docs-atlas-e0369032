  <h2 className="text-2xl font-semibold text-gray-900 dark:text-white mb-3">
    Quick Start
  </h2>
  <p className="text-base text-gray-600 dark:text-gray-400 mb-8">
    Get up and running with Lexa in three simple steps
  </p>
  
  <Steps>
    <Step title="Install the package">
      Install the Lexa SDK using your preferred package manager:
      
      <CodeGroup>
      ```bash npm
      npm install @robilabs/lexa
      ```
      ```bash yarn
      yarn add @robilabs/lexa
      ```
      ```bash pnpm
      pnpm add @robilabs/lexa
      ```
      </CodeGroup>
    </Step>
    
    <Step title="Get your API key">
      Visit [Lexa Chat](https://www.lexa.chat) to create an account and generate your API key from the account settings.
    </Step>
    
    <Step title="Make your first request">
      Initialize the SDK and start making requests:
      
      ```javascript
      import Lexa from '@robilabs/lexa';
      
      const lexa = new Lexa('your-api-key');
      
      const completion = await lexa.chat({
        messages: [
          { role: 'user', content: 'Hello! How are you?' }
        ],
        model: 'lexa-mml',
        temperature: 0.7
      });
      
      console.log(completion.choices[0].message.content);
      ```
    </Step>
  </Steps>
</div>
